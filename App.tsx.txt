const App: React.FC = () => {
  const [customProducts, setCustomProducts] = useState<Product[]>(() => {
    try {
      const saved = localStorage.getItem('pharma_custom_products');
      return saved ? JSON.parse(saved) : [];
    } catch { return []; }
  });
  
  const [showToast, setShowToast] = useState<string | null>(null);
  const [showCartOnly, setShowCartOnly] = useState(false);
  const [quantities, setQuantities] = useState<Record<string, number>>(() => {
    try {
      const saved = localStorage.getItem('pharma_quantities');
      return saved ? JSON.parse(saved) : {};
    } catch { return {}; }
  });

  const [searchQuery, setSearchQuery] = useState('');
  const [isAddingProduct, setIsAddingProduct] = useState(false);
  const [showSyncModal, setShowSyncModal] = useState(false);
  const [viewingSource, setViewingSource] = useState<string | null>(null);
  const [isGeneratingPDF, setIsGeneratingPDF] = useState(false);
  const fileInputRef = useRef<HTMLInputElement>(null);
  
  const [newName, setNewName] = useState('');
  const [newNrv, setNewNrv] = useState('');

  useEffect(() => {
    localStorage.setItem('pharma_custom_products', JSON.stringify(customProducts));
    localStorage.setItem('pharma_quantities', JSON.stringify(quantities));
  }, [customProducts, quantities]);

  useEffect(() => {
    if (showToast) {
      const timer = setTimeout(() => setShowToast(null), 3000);
      return () => clearTimeout(timer);
    }
  }, [showToast]);

  const handleEraseAll = () => {
    if (!confirm("Clear this entire order?")) return;
    setQuantities({});
    setShowCartOnly(false);
    setShowToast("Cleared");
  };

  const updateQuantity = useCallback((id: string, delta: number) => {
    setQuantities(prev => {
      const current = prev[id] || 0;
      const next = Math.max(0, current + delta);
      if (next === 0) {
        const { [id]: _, ...rest } = prev;
        return rest;
      }
      return { ...prev, [id]: next };
    });
  }, []);

  const handleManualInput = (id: string, value: string) => {
    const num = parseInt(value, 10);
    setQuantities(prev => {
      if (isNaN(num) || num <= 0) {
        const { [id]: _, ...rest } = prev;
        return rest;
      }
      return { ...prev, [id]: num };
    });
  };

  const allProducts = useMemo(() => [...STATIC_PRODUCTS, ...customProducts], [customProducts]);
  
  const filteredProducts = useMemo(() => {
    let list = allProducts;
    if (showCartOnly) list = list.filter(p => (quantities[p.id] || 0) > 0);
    const query = searchQuery.toLowerCase();
    return list.filter(p => p.name.toLowerCase().includes(query));
  }, [allProducts, searchQuery, showCartOnly, quantities]);

  const activeItemsCount = Object.keys(quantities).length;
  const grandTotal = allProducts.reduce((acc, p) => acc + (p.nrv * (quantities[p.id] || 0)), 0);

  const generatePDF = async () => {
    const active = allProducts.filter(p => (quantities[p.id] || 0) > 0);
    if (active.length === 0) return alert("Add items first");
    setIsGeneratingPDF(true);
    try {
      const doc = new jsPDF();
      doc.setFontSize(22); doc.setTextColor(67, 56, 202);
      doc.text('PROXIMA ORDER', 14, 20);
      doc.setFontSize(10); doc.setTextColor(100);
      doc.text(`Date: ${new Date().toLocaleString()}`, 14, 28);
      autoTable(doc, {
        startY: 35,
        head: [['Item', 'Price', 'Qty', 'Total']],
        body: active.map(p => [p.name, `â‚¹${p.nrv}`, quantities[p.id], `â‚¹${p.nrv * quantities[p.id]}`]),
        foot: [['', '', 'TOTAL', `â‚¹${grandTotal}`]],
        headStyles: { fillColor: [67, 56, 202] }
      });
      doc.save('Proxima_Order.pdf');
      setShowToast("PDF Saved");
    } finally { setIsGeneratingPDF(false); }
  };

    const active = allProducts.filter(p => (quantities[p.id] || 0) > 0);

    const msg = `ðŸ§¾ *PROXIMA ORDER*\n\n${list}\n\n*TOTAL: â‚¹${grandTotal.toLocaleString()}*`;
    window.open(`https://wa.me/?text=${encodeURIComponent(msg)}`, '_blank');
  };

    <div className="min-h-screen bg-slate-100 flex flex-col font-sans select-none overflow-x-hidden">

                <p className="text-[9px] font-black text-indigo-300 uppercase">Calculator</p>
              </div>
            </div>
            <div className="flex gap-2">
              <button onClick={handleEraseAll} className="bg-white/10 p-2.5 rounded-2xl active:bg-red-500 transition-colors"><Trash2 size={20}/></button>
              <button onClick={() => setIsAddingProduct(!isAddingProduct)} className={`p-2.5 rounded-2xl transition-all ${isAddingProduct ? 'bg-white text-indigo-700' : 'bg-white/10'}`}><PlusCircle size={20}/></button>
              <button onClick={() => setShowSyncModal(true)} className="p-2.5 bg-white/10 rounded-2xl active:bg-white/20"><Smartphone size={20}/></button>
            </div>
          </div>
          
          <div className="flex gap-2">

              <Search className="absolute left-4 top-1/2 -translate-y-1/2 text-indigo-300" size={18} />

            </div>
            <button onClick={() => setShowCartOnly(!showCartOnly)} className={`w-14 h-14 rounded-2xl flex items-center justify-center border-2 transition-all ${showCartOnly ? 'bg-white text-indigo-700 border-white' : 'bg-indigo-800/40 border-white/10'}`}>
              <ShoppingBag size={22} />
            </button>
          </div>
        </div>
      </header>

      <main className="flex-1 max-w-xl mx-auto w-full px-3 py-4 pb-48 space-y-3">
        {isAddingProduct && (
          <div className="bg-white rounded-[2rem] p-6 shadow-2xl border-2 border-indigo-500 animate-in fade-in slide-in-from-top-4 duration-300">
            <h4 className="text-[10px] font-black uppercase text-indigo-500 mb-4 tracking-widest">Add New Medicine</h4>

            <div className="flex gap-2">
              <input type="number" placeholder="PRICE â‚¹" className="flex-1 bg-slate-50 p-4 rounded-xl font-black border-2 border-slate-100" value={newNrv} onChange={e => setNewNrv(e.target.value)} />
              <button onClick={() => {
                const n = parseFloat(newNrv);
                if (newName && !isNaN(n)) {
                  setCustomProducts([{id: `c-${Date.now()}`, name: newName.toUpperCase(), nrv: n}, ...customProducts]);
                  setNewName(''); setNewNrv(''); setIsAddingProduct(false);
                  setShowToast("Added Product");
                }
              }} className="bg-indigo-600 text-white px-8 rounded-xl font-black active:scale-95 transition-transform">ADD</button>
            </div>
          </div>
        )}

        <div className="grid grid-cols-1 gap-2.5">
          {filteredProducts.map(p => (
            <div key={p.id} className={`bg-white rounded-[1.75rem] p-4 border-2 transition-all duration-200 ${quantities[p.id] ? 'border-indigo-500 shadow-md ring-4 ring-indigo-500/5' : 'border-slate-200/60'}`}>
              <div className="flex justify-between items-start mb-3">
                <div className="flex-1">
                  <h3 className={`text-sm font-black uppercase ${quantities[p.id] ? 'text-indigo-900' : 'text-slate-800'}`}>{p.name}</h3>
                  <p className="text-[10px] font-bold text-slate-400 mt-1">MRP â‚¹{p.nrv}</p>
                </div>
              </div>
              <div className="flex items-center justify-between gap-4">
                <div className="flex items-center bg-slate-100 rounded-2xl p-1 gap-1 flex-1 max-w-[180px]">
                  <button onClick={() => updateQuantity(p.id, -1)} className="w-11 h-11 flex items-center justify-center bg-white rounded-xl shadow-sm text-red-500 active:scale-90 transition-transform"><Minus size={20}/></button>
                  <input type="number" className="w-full bg-transparent text-center font-black text-xl outline-none" value={quantities[p.id] || ''} onChange={e => handleManualInput(p.id, e.target.value)} placeholder="0" />
                  <button onClick={() => updateQuantity(p.id, 1)} className="w-11 h-11 flex items-center justify-center bg-indigo-600 text-white rounded-xl active:scale-90 transition-transform"><Plus size={20}/></button>
                </div>
                <div className="text-right min-w-[80px]">

                    <p className="text-lg font-black text-indigo-700 animate-in zoom-in duration-200">â‚¹{p.nrv * quantities[p.id]}</p>

                    </div>

            <div className="py-20 text-center space-y-4">
              <div className="w-20 h-20 bg-slate-200 rounded-full flex items-center justify-center mx-auto opacity-50">
                <Search size={32} className="text-slate-400" />
              </div>
              <p className="font-black text-slate-400 uppercase text-xs tracking-widest">No matching medicines</p>
            </div>
          )}
        </div>
      </main>

      <footer className="fixed bottom-0 left-0 right-0 p-4 z-40">
        <div className="max-w-xl mx-auto bg-slate-900 text-white rounded-[2.5rem] p-4 flex items-center justify-between shadow-2xl border border-white/10 backdrop-blur-lg">
          <div className="pl-4">
            <span className="text-[10px] opacity-40 uppercase font-black tracking-widest">{activeItemsCount} Items Selected</span>
            <p className="text-3xl font-black text-indigo-300">â‚¹{grandTotal.toLocaleString()}</p>
          </div>
          <div className="flex gap-2">
            <button onClick={shareWhatsApp} className="w-14 h-14 bg-green-500 rounded-2xl flex items-center justify-center active:scale-90 shadow-lg shadow-green-500/20 transition-transform"><MessageCircle size={24}/></button>
            <button onClick={generatePDF} disabled={isGeneratingPDF} className={`w-14 h-14 bg-rose-500 rounded-2xl flex items-center justify-center active:scale-90 shadow-lg shadow-rose-500/20 transition-transform ${isGeneratingPDF ? 'opacity-50' : ''}`}><FileDown size={24}/></button>
            <button onClick={() => setShowSyncModal(true)} className="w-14 h-14 bg-indigo-600 rounded-2xl flex items-center justify-center active:scale-90 shadow-lg shadow-indigo-500/20 transition-transform"><Smartphone size={24}/></button>
          </div>
        </div>
      </footer>

      {showSyncModal && (
        <div className="fixed inset-0 z-[100] bg-black/80 flex items-center justify-center p-6 backdrop-blur-md animate-in fade-in duration-300">
          <div className="bg-white rounded-[2.5rem] w-full max-w-md overflow-hidden flex flex-col shadow-2xl max-h-[90vh] animate-in slide-in-from-bottom-8 duration-300">
            <div className="bg-indigo-700 p-8 text-white text-center relative">
              <button onClick={() => setShowSyncModal(false)} className="absolute top-6 right-6 p-2 bg-white/10 rounded-full hover:bg-white/20 transition-colors"><X size={20}/></button>
              <div className="flex flex-col items-center gap-2">
                <Laptop size={32} className="text-indigo-200" />
                <h2 className="text-2xl font-black uppercase tracking-tight">Laptop Settings</h2>
              </div>

                  onClick={() => setViewingSource('App.tsx')}
                  className="w-full flex items-center justify-between p-4 bg-white text-indigo-700 rounded-2xl font-black text-xs border border-indigo-200 active:scale-95 transition-all shadow-sm"
                >

                  <ChevronRight size={18}/>
                </button>
                <p className="mt-3 text-[9px] text-slate-500 font-bold leading-relaxed px-1 uppercase tracking-tight">
                  Click above to copy files to your laptop and upload to GitHub/Vercel.

                <button onClick={() => {
                  const data = JSON.stringify({ q: quantities, c: customProducts });
                  const blob = new Blob([data], { type: 'application/json' });
                  const url = URL.createObjectURL(blob);
                  const a = document.createElement('a'); a.href = url; a.download = 'Proxima_Backup.json'; a.click();
                }} className="flex flex-col items-center justify-center p-5 bg-slate-50 rounded-3xl font-black text-[9px] gap-2 border border-slate-200 hover:bg-white transition-colors">
                  <Download size={20} className="text-indigo-600"/> EXPORT BACKUP
                </button>
                <button onClick={() => fileInputRef.current?.click()} className="flex flex-col items-center justify-center p-5 bg-slate-50 rounded-3xl font-black text-[9px] gap-2 border border-slate-200 hover:bg-white transition-colors">
                  <Upload size={20} className="text-indigo-600"/> IMPORT BACKUP
                </button>
              </div>

              <input type="file" ref={fileInputRef} className="hidden" onChange={(e) => {
                const file = e.target.files?.[0];
                if (!file) return;
                const reader = new FileReader();
                reader.onload = (ev) => {
                  try {
                    const d = JSON.parse(ev.target?.result as string);
                    if (d.q) setQuantities(d.q);
                    if (d.c) setCustomProducts(d.c);
                    setShowToast("Restored Successfully");
                    setShowSyncModal(false);
                  } catch { alert("Invalid Backup File Format"); }
                };
                reader.readAsText(file);
              }} />

              <button onClick={() => setShowSyncModal(false)} className="w-full py-5 bg-slate-900 text-white font-black rounded-2xl uppercase text-sm tracking-widest active:scale-95 transition-transform">Close Menu</button>
            </div>

          <div className="flex justify-between items-center mb-4">

              </div>
              <button

            </div>
            <div className="flex-1 p-5 font-mono text-[10px] overflow-auto whitespace-pre leading-relaxed selection:bg-indigo-500/50 no-scrollbar">

            </div>
          </div>

          <div className="space-y-3">

                const content = PROJECT_FILES[viewingSource];
                navigator.clipboard.writeText(content);
                setShowToast("Ready to Paste");
              }}
              className="w-full py-5 bg-indigo-600 rounded-2xl font-black flex items-center justify-center gap-3 active:scale-95 shadow-xl transition-all"
            >
              <Copy size={20}/> COPY {viewingSource}
            </button>
            <p className="text-[10px] text-center text-slate-500 font-bold uppercase tracking-[0.2em] leading-loose">

            </p>
          </div>
        </div>
      )}
    </div>
  );


import React, { useState, useMemo } from 'react';
import { 
  Search, Trash2, Calculator, ShoppingBag, 
  Plus, Minus, FileDown, MessageCircle
} from 'lucide-react';
import { jsPDF } from 'jspdf';
import autoTable from 'jspdf-autotable';
import { PRODUCTS as STATIC_PRODUCTS } from './constants';

/**
 * Main application component for the Proxima Order Calculator.
 * Resolves the default export error in index.tsx.
 */
const App: React.FC = () => {
  const [quantities, setQuantities] = useState<Record<string, number>>({});
  const [searchQuery, setSearchQuery] = useState('');
  const [isDownloading, setIsDownloading] = useState(false);

  // Filter products based on search query
  const filteredProducts = useMemo(() => {
    return STATIC_PRODUCTS.filter(product =>
      product.name.toLowerCase().includes(searchQuery.toLowerCase())
    );
  }, [searchQuery]);

  // Update product quantity in cart
  const updateQuantity = (id: string, delta: number) => {
    setQuantities(prev => ({
      ...prev,
      [id]: Math.max(0, (prev[id] || 0) + delta)
    }));
  };

  // Clear all items from the cart
  const clearCart = () => {
    if (window.confirm('Are you sure you want to clear all items?')) {
      setQuantities({});
    }
  };

  // Derive cart items from quantities state
  const cartItems = useMemo(() => {
    return Object.entries(quantities)
      .filter(([_, qty]) => qty > 0)
      .map(([id, qty]) => {
        const product = STATIC_PRODUCTS.find(p => p.id === id);
        return { ...product!, quantity: qty };
      });
  }, [quantities]);

  // Calculate grand total of all items
  const totalAmount = useMemo(() => {
    return cartItems.reduce((sum, item) => sum + (item.nrv * item.quantity), 0);
  }, [cartItems]);

  // Generate and download order summary as PDF
  const generatePDF = () => {
    if (cartItems.length === 0) return;
    setIsDownloading(true);
    try {
      const doc = new jsPDF();
      
      doc.setFontSize(20);
      doc.text('Proxima Order Summary', 14, 22);
      
      doc.setFontSize(11);
      doc.setTextColor(100);
      doc.text(`Generated on: ${new Date().toLocaleString()}`, 14, 30);

      const tableData = cartItems.map(item => [
        item.name,
        item.nrv.toFixed(2),
        item.quantity.toString(),
        (item.nrv * item.quantity).toFixed(2)
      ]);

      autoTable(doc, {
        startY: 40,
        head: [['Product Name', 'NRV', 'Qty', 'Total']],
        body: tableData,
        foot: [['', '', 'Total Amount', `₹${totalAmount.toFixed(2)}`]],
        theme: 'grid',
        headStyles: { fillColor: [67, 56, 202] }
      });

      doc.save('proxima_order.pdf');
    } catch (error) {
      console.error('Error generating PDF:', error);
      alert('Failed to generate PDF. Please try again.');
    } finally {
      setIsDownloading(false);
    }
  };

  // Export order summary to WhatsApp
  const copyToWhatsApp = () => {
    if (cartItems.length === 0) return;
    
    const text = `*Order Summary*\n` +
      cartItems.map(item => `${item.name}: ${item.quantity} x ₹${item.nrv} = ₹${(item.nrv * item.quantity).toFixed(2)}`).join('\n') +
      `\n\n*Total Amount: ₹${totalAmount.toFixed(2)}*`;
    
    const url = `https://wa.me/?text=${encodeURIComponent(text)}`;
    window.open(url, '_blank');
  };

  return (
    <div className="min-h-screen bg-slate-50 font-sans pb-24 lg:pb-0">
      <header className="bg-indigo-700 text-white p-4 sticky top-0 z-10 shadow-lg">
        <div className="max-w-7xl mx-auto flex items-center justify-between">
          <div className="flex items-center gap-3">
            <div className="bg-white/20 p-2 rounded-lg backdrop-blur-sm">
              <Calculator className="w-6 h-6" />
            </div>
            <div>
              <h1 className="text-xl font-bold tracking-tight leading-none">PROXIMA</h1>
              <p className="text-[10px] uppercase tracking-widest font-semibold opacity-70">Order Calculator</p>
            </div>
          </div>
          <button 
            onClick={clearCart}
            className="p-2 hover:bg-white/10 rounded-full transition-colors"
            title="Clear All"
          >
            <Trash2 className="w-5 h-5" />
          </button>
        </div>
      </header>

      <main className="max-w-7xl mx-auto p-4 lg:grid lg:grid-cols-3 lg:gap-8 lg:items-start">
        <div className="lg:col-span-2 space-y-4">
          <div className="relative">
            <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-5 h-5 text-slate-400" />
            <input 
              type="text" 
              placeholder="Search products..."
              className="w-full pl-10 pr-4 py-3 bg-white rounded-xl border-none shadow-sm focus:ring-2 focus:ring-indigo-500 transition-all outline-none text-slate-700"
              value={searchQuery}
              onChange={(e) => setSearchQuery(e.target.value)}
            />
          </div>

          <div className="grid grid-cols-1 md:grid-cols-2 gap-3">
            {filteredProducts.map(product => (
              <div key={product.id} className="bg-white p-4 rounded-xl shadow-sm border border-slate-100 flex items-center justify-between hover:shadow-md transition-shadow">
                <div className="flex-1">
                  <h3 className="font-semibold text-slate-800 text-sm leading-tight">{product.name}</h3>
                  <p className="text-indigo-600 font-bold mt-1 text-sm">₹{product.nrv}</p>
                </div>
                <div className="flex items-center gap-1 bg-slate-50 p-1 rounded-lg border border-slate-200 ml-4">
                  <button 
                    onClick={() => updateQuantity(product.id, -1)}
                    className="p-1.5 hover:bg-white rounded-md transition-colors text-slate-500 hover:text-red-500"
                  >
                    <Minus className="w-4 h-4" />
                  </button>
                  <span className="w-8 text-center font-bold text-slate-700 tabular-nums">
                    {quantities[product.id] || 0}
                  </span>
                  <button 
                    onClick={() => updateQuantity(product.id, 1)}
                    className="p-1.5 hover:bg-white rounded-md transition-colors text-slate-500 hover:text-indigo-600"
                  >
                    <Plus className="w-4 h-4" />
                  </button>
                </div>
              </div>
            ))}
          </div>
        </div>

        <div className="hidden lg:block lg:sticky lg:top-24 bg-white p-6 rounded-2xl shadow-xl border border-slate-100 space-y-6">
          <div className="flex items-center justify-between">
            <h2 className="text-lg font-bold text-slate-800 flex items-center gap-2">
              <ShoppingBag className="w-5 h-5 text-indigo-600" />
              Order Summary
            </h2>
            <span className="bg-indigo-100 text-indigo-700 px-3 py-1 rounded-full text-xs font-bold">
              {cartItems.length} Items
            </span>
          </div>

          <div className="space-y-3 max-h-[400px] overflow-y-auto pr-2 custom-scrollbar">
            {cartItems.map(item => (
              <div key={item.id} className="flex justify-between items-start py-2 border-b border-slate-50 last:border-0">
                <div className="flex-1 pr-4">
                  <p className="text-sm font-medium text-slate-700 leading-tight">{item.name}</p>
                  <p className="text-xs text-slate-400 mt-0.5">{item.quantity} × ₹{item.nrv}</p>
                </div>
                <p className="font-bold text-slate-800 text-sm">₹{(item.quantity * item.nrv).toFixed(2)}</p>
              </div>
            ))}
            {cartItems.length === 0 && (
              <div className="text-center py-12 text-slate-400">
                <ShoppingBag className="w-12 h-12 mx-auto mb-3 opacity-20" />
                <p>No items added yet</p>
              </div>
            )}
          </div>

          <div className="pt-6 border-t border-slate-100">
            <div className="flex justify-between items-center mb-6">
              <span className="text-slate-500 font-medium">Grand Total</span>
              <span className="text-2xl font-black text-slate-900 tracking-tight">₹{totalAmount.toFixed(2)}</span>
            </div>

            <div className="grid grid-cols-2 gap-3">
              <button 
                onClick={generatePDF}
                disabled={cartItems.length === 0 || isDownloading}
                className="flex items-center justify-center gap-2 bg-indigo-600 hover:bg-indigo-700 disabled:bg-slate-200 text-white py-3 rounded-xl font-bold transition-all shadow-md"
              >
                <FileDown className="w-5 h-5" />
                PDF
              </button>
              <button 
                onClick={copyToWhatsApp}
                disabled={cartItems.length === 0}
                className="flex items-center justify-center gap-2 bg-emerald-500 hover:bg-emerald-600 disabled:bg-slate-200 text-white py-3 rounded-xl font-bold transition-all shadow-md"
              >
                <MessageCircle className="w-5 h-5" />
                WhatsApp
              </button>
            </div>
          </div>
        </div>
      </main>

      {/* Mobile Sticky Footer */}
      <div className="lg:hidden fixed bottom-0 left-0 right-0 p-4 bg-white border-t border-slate-100 shadow-[0_-8px_30px_rgb(0,0,0,0.04)] z-20">
        <div className="max-w-lg mx-auto flex items-center justify-between gap-4">
          <div className="flex-1">
            <p className="text-[10px] text-slate-400 font-bold uppercase tracking-wider">Total Amount</p>
            <p className="text-xl font-black text-slate-800 tracking-tight leading-none">₹{totalAmount.toFixed(2)}</p>
          </div>
          <button 
            onClick={generatePDF}
            disabled={cartItems.length === 0 || isDownloading}
            className="bg-indigo-600 hover:bg-indigo-700 disabled:bg-slate-200 text-white h-12 px-6 rounded-xl font-bold flex items-center gap-2 transition-all shadow-lg shadow-indigo-200"
          >
            {isDownloading ? '...' : (
              <>
                <FileDown className="w-5 h-5" />
                Checkout
              </>
            )}
          </button>
        </div>
      </div>
    </div>
  );
};

export default App;
